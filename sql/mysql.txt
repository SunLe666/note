先停止mysql服务
找到my.ini文件
修改datadir=C:/ProgramData/MySQL/MySQL Server 5.5\data

use information_schema;

select table_name,table_rows from tables

where TABLE_SCHEMA = '数据库名称'

order by table_rows asc;
